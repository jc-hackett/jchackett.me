(function menuController() {
  const options = Array.from(document.querySelectorAll(".menu-option"));
  const panels = Array.from(document.querySelectorAll(".menu-panel"));

  if (!options.length || !panels.length) return;

  function setActivePanel(index) {
    options.forEach((option, i) => {
      const isActive = i === index;
      option.classList.toggle("active", isActive);
      option.setAttribute("aria-selected", String(isActive));
    });

    panels.forEach((panel, i) => {
      const isActive = i === index;
      panel.classList.toggle("active", isActive);
      panel.hidden = !isActive;
    });
  }

  const card = document.querySelector('.card');
const hour = new Date().getHours();

let x = 0;
let y = 12;

if (hour < 12) {
  x = -10;  // morning light from left
}
else if (hour < 17) {
  x = 0;    // midday overhead
}
else {
  x = 10;   // evening light from right
}

card.style.boxShadow = `${x}px ${y}px 28px rgba(0,0,0,0.18)`;

  options.forEach((option, index) => {
    option.addEventListener("click", function () {
      const isAlreadyOpen = option.classList.contains("active");

      if (isAlreadyOpen) {
        options.forEach((btn) => {
          btn.classList.remove("active");
          btn.setAttribute("aria-selected", "false");
        });

        panels.forEach((panel) => {
          panel.classList.remove("active");
          panel.hidden = true;
        });

        return;
      }

      setActivePanel(index);
    });
  });

  panels.forEach((panel, index) => {
    panel.hidden = !panel.classList.contains("active");
    if (panel.classList.contains("active")) {
      options[index]?.classList.add("active");
      options[index]?.setAttribute("aria-selected", "true");
    }
  });
})();

LOG_FILE="$KERNEL_DIR/kernel-build.log"

exec > >(tee -a "$LOG_FILE") 2>&1

echo "--------------------------------"
echo "Kernel build started: $(date)"
echo "Project root: $PROJECT_ROOT"
echo "Log file: $LOG_FILE"
echo "--------------------------------"

#!/usr/bin/env bash
set -euo pipefail
trap 'echo "ERROR on line $LINENO"; exit 1' ERR

MAX_BYTES=50000

PROJECT_ROOT="$(pwd)"
ERASMUS_DIR="$PROJECT_ROOT/ERASMUS"
KERNEL_DIR="$ERASMUS_DIR/_kernels"

mkdir -p "$KERNEL_DIR"

OUT_FILE="$KERNEL_DIR/kernel-$(date +%Y%m%d-%H%M%S).md"

echo "Starting Erasmus kernel build..."
echo "Writing kernel to $OUT_FILE"

current_size() {
  wc -c < "$OUT_FILE" | tr -d ' '
}

can_fit() {
  local add_bytes="$1"
  local now
  now="$(current_size)"
  [ $((now + add_bytes)) -le "$MAX_BYTES" ]
}

append_text() {
  local text="$1"
  local bytes
  bytes=$(printf "%s" "$text" | wc -c | tr -d ' ')
  if can_fit "$bytes"; then
    printf "%s" "$text" >> "$OUT_FILE"
    return 0
  fi
  return 1
}

append_file() {
  local file="$1"
  local rel="${file#$PROJECT_ROOT/}"
  local tmp
  tmp="$(mktemp)"

  {
    printf "## FILE: %s\n\n" "$rel"
    printf '```\n'
    sed 's/\r$//' "$file"
    printf '\n```\n\n'
  } > "$tmp"

  local bytes
  bytes=$(wc -c < "$tmp" | tr -d ' ')

  if can_fit "$bytes"; then
    echo "Exporting $rel"
    cat "$tmp" >> "$OUT_FILE"
  else
    echo "Skipping $rel (size cap)"
    append_text "## SKIPPED: $rel\n\nSkipped to keep kernel under ${MAX_BYTES} bytes.\n\n"
  fi

  rm -f "$tmp"
}

is_text_file() {
  case "$1" in
    *.md|*.njk|*.html|*.css|*.js|*.json|*.toml|*.yml|*.yaml|*.sh|*.cmd|*.txt)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

{
  echo "# Erasmus Kernel Snapshot"
  echo "Generated: $(date)"
  echo "Project root: $PROJECT_ROOT"
  echo
  echo "Max size: ${MAX_BYTES} bytes"
  echo
} > "$OUT_FILE"

append_text "## DIRECTORY STRUCTURE\n\n\`\`\`\n"

find . \
  -path "./node_modules" -prune -o \
  -path "./_site" -prune -o \
  -path "./.git" -prune -o \
  -path "./ERASMUS/_kernels" -prune -o \
  -print | sed 's#^\./##' | while IFS= read -r line; do
    append_text "$line"$'\n' || break
  done

append_text "\`\`\`\n\n"

echo "Exporting src..."
find "$PROJECT_ROOT/src" -type f | sort | while IFS= read -r file; do
  is_text_file "$file" && append_file "$file"
done

echo "Exporting ERASMUS..."
find "$PROJECT_ROOT/ERASMUS" \
  -path "$PROJECT_ROOT/ERASMUS/_kernels" -prune -o \
  -type f -print | sort | while IFS= read -r file; do
    is_text_file "$file" && append_file "$file"
done

echo "Exporting root config..."
for file in \
  "$PROJECT_ROOT/.eleventy.js" \
  "$PROJECT_ROOT/.eleventyignore" \
  "$PROJECT_ROOT/netlify.toml" \
  "$PROJECT_ROOT/package.json" \
  "$PROJECT_ROOT/package-lock.json" \
  "$PROJECT_ROOT/.gitignore" \
  "$PROJECT_ROOT/README.md"
do
  [ -f "$file" ] && append_file "$file"
done

append_text "---\n\nKernel complete: $(basename "$OUT_FILE")\n"

echo
echo "Kernel written:"
echo "$OUT_FILE"
echo "Size: $(current_size) bytes"

Keep a running note of all side items directed by user. When asked to take a note or mark something, just mark it. Add it to feedback.

The project uses a simple, layered structure that separates site content, reusable components, styling, and process infrastructure. All site source files live inside `src`, which is Eleventy’s input directory. Layouts are in `src/_layouts`, reusable interface components live in `src/_includes/components`, and page content such as `index.md` sits at the root of `src`. Styling is organized under `src/css`, with component-specific styles placed in `src/css/components`. The disclosure interface is implemented through `disclosure-group.njk` and styled by `disclosure.css`, while `services.njk` and `contact.njk` provide content wrappers. Outside `src`, the `ERASMUS` directory stores kernel snapshots, process documentation, and rebuild scripts that preserve project memory between development iterations.

Project has moved from a cloned “menu” pattern to a coherent disclosure system. Current architecture is clean: `disclosure-group.njk` owns structure, `disclosure.css` owns styling, and `services.njk` / `contact.njk` act as content wrappers. Naming consistency materially improved readability and reduced mental context switching. Main lesson from this round: local visual debugging stayed manageable when changes were kept scoped and ownership boundaries were clarified rather than expanded. Future work should preserve this separation. Do not reintroduce duplicate component markup. When adjusting spacing or behavior, change the shared disclosure layer first. Verify `base.njk` stylesheet links whenever a component CSS file is renamed.

The site is designed to present information calmly and clearly, without visual clutter or navigational complexity. Visitors arrive on a single page where the essential introduction appears immediately, followed by structured sections that reveal additional information only when requested. The disclosure interface allows users to expand topics such as services or contact details at their own pace, keeping the page visually quiet while still providing depth. Typography, spacing, and restrained color choices emphasize readability and focus. The experience should feel deliberate, respectful of the visitor’s attention, and easy to navigate on both desktop and mobile, allowing users to quickly understand the purpose of the site and decide whether to reach out.

The most effective collaboration happens when instructions are concrete, scoped, and tied to specific files. When requesting changes, specify the file, the goal, and whether the task is structural, visual, or diagnostic. During visual debugging, small targeted changes are best; avoid asking for broad redesigns unless that is the explicit goal. When possible, paste the current contents of the file being modified so the assistant can work from the exact state of the project. Indicating whether you want explanation, implementation, or review also helps. Maintaining kernel snapshots and clear component ownership further improves continuity, allowing future iterations of the assistant to understand the project quickly and provide precise guidance.
